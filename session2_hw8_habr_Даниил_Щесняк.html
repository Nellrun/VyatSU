def toVWString(d, argName):
    elem = d[argName]
    if type(elem) == list:
        if (len(elem) == 0):
            return "|" + argName + " "
        if type(elem[0]) != str:
            elem = map(str, elem)
        return "|" + argName + " " + " ".join(elem) + " "
    if type(elem) == dict:
        s = ""
        for key in elem.keys():
            s += key + ":" + str(elem[key]) + " "
        return "|" + argName + " " + s
    return "|" + argName + " " + str(elem) + " "

def dictToVWString(d, l=None, value=None):
    if l == None:
        l = d.keys()
        
    out = "" if value == None else str(value) + " "
    for elem in l:
        out += toVWString(d, elem)
    return out
